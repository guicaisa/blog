    string longestCommonPrefix(vector<string>& strs) 
{
    string result;
    for (int i = 0; i < strs[0].size(); ++i)
    {
        char ch = strs[0][i];
        bool end = false;
        for (int j = 1; j < strs.size(); ++j)
        {
            //注意边界情况，strs[0]的长度可能比其他字符串长
            if (i >= strs[j].size() || strs[j][i] != ch)
            {
                end = true;
                break;
            }
        }
        if (end)
        {
            break;
        }
        result += ch;
    }

    return result;
}
